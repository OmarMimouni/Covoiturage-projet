        async function chargerStatistiques() {
            try {
                // Récupérer le nombre total de réservations
                const { count: totalReservations, error: reservationsError } = await supabase
                    .from('reservations')
                    .select('*', { count: 'exact', head: true });
                
                if (reservationsError) throw reservationsError;
                
                // Récupérer le nombre total d'utilisateurs
                const { count: totalUtilisateurs, error: usersError } = await supabase
                    .from('utilisateurs')
                    .select('*', { count: 'exact', head: true });
                
                if (usersError) throw usersError;
                
                // Récupérer les réservations validées
                const { count: reservationsValidees, error: valideesError } = await supabase
                    .from('reservations')
                    .select('*', { count: 'exact', head: true })
                    .eq('valide_conducteur', true);
                
                if (valideesError) throw valideesError;
                
                // Calculer le revenu total
                const { data: reservationsRevenu, error: revenuError } = await supabase
                    .from('reservations')
                    .select('prix_reservation')
                    .eq('valide_conducteur', true);

                document.getElementById('total-utilisateurs').textContent = totalUtilisateurs;
                document.getElementById('reservations-validees').textContent = reservationsValidees;
                document.getElementById('revenu-total').textContent = `${revenuTotal.toFixed(2)} MAD`;

                console.error('Erreur lors du chargement des statistiques:', error);

        // Charger les graphiques
        async function chargerGraphiques() {
            try {
                // Récupérer les réservations des 7 derniers jours
                const dateLimite = new Date();
                dateLimite.setDate(dateLimite.getDate() - 7);
                
                const { data: reservations, error } = await supabase
                    .from('reservations')
                    .select('created_at')
                    .gte('created_at', dateLimite.toISOString());
                
                if (error) throw error;
                
                // Préparer les données pour le graphique des réservations par jour
                const reservationsParJour = {};
                
                // Initialiser les 7 derniers jours avec 0
                for (let i = 6; i >= 0; i--) {
                    const date = new Date();
                    date.setDate(date.getDate() - i);
                    const dateStr = date.toISOString().split('T')[0]; // Format YYYY-MM-DD
                    reservationsParJour[dateStr] = 0;
                }
                
                // Compter les réservations par jour
                reservations.forEach(res => {
                    const date = new Date(res.created_at);
                    const dateStr = date.toISOString().split('T')[0]; // Format YYYY-MM-DD
                    
                    if (reservationsParJour[dateStr] !== undefined) {
                        reservationsParJour[dateStr]++;
                    }
                });
                
                // Formater les dates pour l'affichage (JJ/MM)
                const labels = Object.keys(reservationsParJour).map(dateStr => {
                    const date = new Date(dateStr);
                    return `${date.getDate().toString().padStart(2, '0')}/${(date.getMonth() + 1).toString().padStart(2, '0')}`;
                });
                
                const data = Object.values(reservationsParJour);
                
                // Créer le graphique des réservations par jour
                const reservationsCtx = document.getElementById('reservationsChart').getContext('2d');
                if (reservationsChart) reservationsChart.destroy();
                
                reservationsChart = new Chart(reservationsCtx, {
                    type: 'bar',
                    data: {
                        labels: labels,
                        datasets: [{
                            label: 'Réservations',
                            data: data,
                            backgroundColor: 'rgba(52, 152, 219, 0.7)',
                            borderColor: 'rgba(52, 152, 219, 1)',
                            borderWidth: 1,
                            borderRadius: 4,
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: {
                                display: false
                            },
                            tooltip: {
                                callbacks: {
                                    label: function(context) {
                                        return `Réservations: ${context.parsed.y}`;
                                    }
                                }
                            }
                        },
                        scales: {
                            y: {
                                beginAtZero: true,
                                ticks: {
                                    stepSize: 1
                                },
                                title: {
                                    display: true,
                                    text: 'Nombre de réservations'
                                }
                            },
                            x: {
                                title: {
                                    display: true,
                                    text: 'Jours'
                                }
                            }
                        }
                    }
                });
                
                // Récupérer les statuts des réservations
                const { data: reservationsStatut, error: statutError } = await supabase
                    .from('reservations')
                    .select('valide_passager, valide_conducteur');
                
                if (statutError) throw statutError;
                
                let enAttente = 0;
                let validees = 0;
                let refusees = 0;
                
                reservationsStatut.forEach(res => {
                    if (res.valide_conducteur) {
                        validees++;
                    } else if (res.valide_passager && !res.valide_conducteur) {
                        enAttente++;
                    } else {
                        refusees++;
                    }
                });
                
                // Créer le graphique des statuts
                const statusCtx = document.getElementById('statusChart').getContext('2d');
                if (statusChart) statusChart.destroy();
                
                statusChart = new Chart(statusCtx, {
                    type: 'doughnut',
                    data: {
                        labels: ['Validées', 'En attente', 'Refusées'],
                        datasets: [{
                            data: [validees, enAttente, refusees],
                            backgroundColor: [
                                'rgba(39, 174, 96, 0.8)',
                                'rgba(243, 156, 18, 0.8)',
                                'rgba(231, 76, 60, 0.8)'
                            ],
                            borderColor: [
                                'rgba(39, 174, 96, 1)',
                                'rgba(243, 156, 18, 1)',
                                'rgba(231, 76, 60, 1)'
                            ],
                            borderWidth: 1
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: {
                                position: 'bottom'
                            },
                            tooltip: {
                                callbacks: {
                                    label: function(context) {
                                        const label = context.label || '';
                                        const value = context.parsed;
                                        const total = context.dataset.data.reduce((a, b) => a + b, 0);
                                        const percentage = Math.round((value / total) * 100);
                                        return `${label}: ${value} (${percentage}%)`;
                                    }
                                }
                            }
                        }
                    }
                });
                
            } catch (error) {
                console.error('Erreur lors du chargement des graphiques:', error);
            }
        }

        // Charger les utilisateurs
        async function chargerUtilisateurs() {
            try {
                const { data: utilisateurs, error } = await supabase
                    .from('utilisateurs')
                    .select('*')
                    .order('nom');
                
                if (error) throw error;
                
                // Récupérer les réservations pour compter le nombre de réservations par utilisateur
                const { data: reservations, error: resError } = await supabase
                    .from('reservations')
                    .select('emailC, emailp');
                
                if (resError) throw resError;
                
                // Compter les réservations par utilisateur
                const reservationsParUtilisateur = {};
                reservations.forEach(res => {
                    // Compter comme conducteur
                    if (!reservationsParUtilisateur[res.emailC]) {
                        reservationsParUtilisateur[res.emailC] = { conducteur: 0, passager: 0 };
                    }
                    reservationsParUtilisateur[res.emailC].conducteur++;
                    
                    // Compter comme passager
                    if (res.emailp && res.emailp.length > 0) {
                        res.emailp.forEach(email => {
                            if (!reservationsParUtilisateur[email]) {
                                reservationsParUtilisateur[email] = { conducteur: 0, passager: 0 };
                            }
                            reservationsParUtilisateur[email].passager++;
                        });
                    }
                });
                
                // Remplir les tables
                const allUsersTable = document.getElementById('all-users-table');
                const conducteursTable = document.getElementById('conducteurs-table');
                const passagersTable = document.getElementById('passagers-table');
                
                allUsersTable.innerHTML = '';
                conducteursTable.innerHTML = '';
                passagersTable.innerHTML = '';
                
                utilisateurs.forEach(user => {
                    const userReservations = reservationsParUtilisateur[user.email] || { conducteur: 0, passager: 0 };
                    const role = userReservations.conducteur > 0 ? 'Conducteur' : 'Passager';
                    const statut = user.online ? 'En ligne' : 'Hors ligne';

                        <td><span class="badge ${user.online ? 'badge-success' : 'badge-warning'}">${statut}</span></td>
                        <td>
                            <button class="btn btn-sm btn-outline-primary me-1" onclick="afficherDetailsUtilisateur('${user.id}')">
                                <i class="bi bi-eye"></i>
                            </button>
                            <button class="btn btn-sm btn-outline-danger" onclick="supprimerUtilisateur('${user.id}', '${user.prenom} ${user.nom}')">
                                <i class="bi bi-trash"></i>
                            </button>
                        </td>
                    `;
                    allUsersTable.appendChild(allUsersRow);
                    
                    // Ligne pour conducteurs si applicable
                    if (userReservations.conducteur > 0) {
                        const conducteursRow = document.createElement('tr');
                        conducteursRow.innerHTML = `
                            <td>${user.prenom} ${user.nom}</td>
                            <td>${user.email}</td>
                            <td>${userReservations.conducteur} trajet(s)</td>
                            <td><span class="badge ${user.online ? 'badge-success' : 'badge-warning'}">${statut}</span></td>
                            <td>
                                <button class="btn btn-sm btn-outline-primary me-1" onclick="afficherDetailsUtilisateur('${user.id}')">
                                    <i class="bi bi-eye"></i>
                                </button>
                                <button class="btn btn-sm btn-outline-danger" onclick="supprimerUtilisateur('${user.id}', '${user.prenom} ${user.nom}')">
                                    <i class="bi bi-trash"></i>
                                </button>
                            </td>
                        `;
                        conducteursTable.appendChild(conducteursRow);
                    }
                    
                    // Ligne pour passagers si applicable
                    if (userReservations.passager > 0) {
                        const passagersRow = document.createElement('tr');
                        passagersRow.innerHTML = `
                            <td>${user.prenom} ${user.nom}</td>
                            <td>${user.email}</td>
                            <td>${userReservations.passager} réservation(s)</td>
                            <td><span class="badge ${user.online ? 'badge-success' : 'badge-warning'}">${statut}</span></td>
                            <td>
                                <button class="btn btn-sm btn-outline-primary me-1" onclick="afficherDetailsUtilisateur('${user.id}')">
                                    <i class="bi bi-eye"></i>
                                </button>
                                <button class="btn btn-sm btn-outline-danger" onclick="supprimerUtilisateur('${user.id}', '${user.prenom} ${user.nom}')">
                                    <i class="bi bi-trash"></i>
                                </button>
                            </td>
                        `;
                        passagersTable.appendChild(passagersRow);
                    }
                });
                
            } catch (error) {
                console.error('Erreur lors du chargement des utilisateurs:', error);
            }
        }

        // Afficher les détails d'un utilisateur
        window.afficherDetailsUtilisateur = async function(userId) {
            try {
                const { data: user, error } = await supabase
                    .from('utilisateurs')
                    .select('*')
                    .eq('id', userId)
                    .single();
                
                if (error) throw error;
                
                // Récupérer les réservations de l'utilisateur
                const { data: reservationsConducteur, error: resConducteurError } = await supabase
                    .from('reservations')
                    .select('*')
                    .eq('emailC', user.email);
                
                if (resConducteurError) throw resConducteurError;
                
                const { data: reservationsPassager, error: resPassagerError } = await supabase
                    .from('reservations')
                    .select('*');
                
                if (resPassagerError) throw resPassagerError;
                
                // Filtrer les réservations où l'utilisateur est passager
                const reservationsCommePassager = reservationsPassager.filter(res => 
                    res.emailp && res.emailp.includes(user.email)
                );
                
                // Préparer le contenu du modal
                const modalBody = document.getElementById('userModalBody');
                modalBody.innerHTML = `
                    <div class="row">
                        <div class="col-md-6">
                            <h6>Informations Personnelles</h6>
                            <table class="table table-sm">
                                <tr>
                                    <td><strong>Nom:</strong></td>
                                    <td>${user.nom}</td>
                                </tr>
                                <tr>
                                    <td><strong>Prénom:</strong></td>
                                    <td>${user.prenom}</td>
                                </tr>
                                <tr>
                                    <td><strong>Email:</strong></td>
                                    <td>${user.email}</td>
                                </tr>
                                <tr>
                                    <td><strong>Statut:</strong></td>
                                    <td><span class="badge ${user.online ? 'badge-success' : 'badge-warning'}">${user.online ? 'En ligne' : 'Hors ligne'}</span></td>
                                </tr>
                            </table>
                        </div>
                        <div class="col-md-6">
                            <h6>Statistiques</h6>
                            <table class="table table-sm">
                                <tr>
                                    <td><strong>Trajets comme conducteur:</strong></td>
                                    <td>${reservationsConducteur.length}</td>
                                </tr>
                                <tr>
                                    <td><strong>Réservations comme passager:</strong></td>
                                    <td>${reservationsCommePassager.length}</td>
                                </tr>
                            </table>
                        </div>
                    </div>
                    
                    ${user.bio_conducteur || user.bio_passager ? `
                    <div class="mt-3">
                        <h6>Biographie</h6>
                        <p>${user.bio_conducteur || user.bio_passager || 'Aucune biographie disponible'}</p>
                    </div>
                    ` : ''}
                    
                    ${reservationsConducteur.length > 0 ? `
                    <div class="mt-3">
                        <h6>Derniers trajets comme conducteur</h6>
                        <div class="table-responsive">
                            <table class="table table-sm">
                                <thead>
                                    <tr>
                                        <th>Départ</th>
                                        <th>Arrivée</th>
                                        <th>Date</th>
                                        <th>Prix</th>
                                    </tr>
                                </thead>
                                <tbody>
                                    ${reservationsConducteur.slice(0, 5).map(res => `
                                        <tr>
                                            <td>${res.debut}</td>
                                            <td>${res.fin}</td>
                                            <td>${new Date(res.created_at).toLocaleDateString()}</td>
                                            <td>${res.prix_reservation} MAD</td>
                                        </tr>
                                    `).join('')}
                                </tbody>
                            </table>
                        </div>
                    </div>
                    ` : ''}
                `;
                
                // Afficher le modal
                const userModal = new bootstrap.Modal(document.getElementById('userModal'));
                userModal.show();
                
            } catch (error) {
                console.error('Erreur lors du chargement des détails utilisateur:', error);
                alert('Erreur lors du chargement des détails utilisateur');
            }
        }

        // Supprimer un utilisateur
        window.supprimerUtilisateur = async function(userId, nomComplet) {
            if (!confirm(`Êtes-vous sûr de vouloir supprimer l'utilisateur "${nomComplet}" ? Cette action est irréversible.`)) {
                return;
            }
            
            try {
                const { error } = await supabase
                    .from('utilisateurs')
                    .delete()
                    .eq('id', userId);
                
                if (error) throw error;
                
                alert('Utilisateur supprimé avec succès !');
                chargerUtilisateurs();
                chargerStatistiques();
                
            } catch (error) {
                console.error('Erreur lors de la suppression:', error);
                alert('Erreur lors de la suppression de l\'utilisateur');
            }
        }

        // Charger les réservations
        async function chargerReservations() {
            try {
                const { data: reservations, error } = await supabase
                    .from('reservations')
                    .select('*')
                    .order('created_at', { ascending: false });
                
                if (error) throw error;
                
                const reservationsTable = document.getElementById('reservations-table');
                reservationsTable.innerHTML = '';
                
                reservations.forEach(res => {
                    let statut, badgeClass;
                    
                    if (res.valide_conducteur) {
                        statut = 'Validée';
                        badgeClass = 'badge-success';
                    } else if (res.valide_passager && !res.valide_conducteur) {
                        statut = 'En attente';
                        badgeClass = 'badge-warning';
                    } else {
                        statut = 'Refusée';
                        badgeClass = 'badge-danger';
                    }
                    
                    const row = document.createElement('tr');
                    row.innerHTML = `
                        <td>${res.id}</td>
                        <td>${res.emailC}</td>
                        <td>${res.debut} → ${res.fin}</td>
                        <td>${new Date(res.created_at).toLocaleDateString()}</td>
                        <td>${res.prix_reservation} MAD</td>
                        <td><span class="badge ${badgeClass}">${statut}</span></td>
                        <td>
                            <button class="btn btn-sm btn-outline-danger" onclick="supprimerReservation(${res.id})">
                                <i class="bi bi-trash"></i>
                            </button>
                        </td>
                    `;
                    reservationsTable.appendChild(row);
                });
                
            } catch (error) {
                console.error('Erreur lors du chargement des réservations:', error);
            }
        }

        // Supprimer une réservation
        window.supprimerReservation = async function(reservationId) {
            if (!confirm('Êtes-vous sûr de vouloir supprimer cette réservation ? Cette action est irréversible.')) {
                return;
            }
            
            try {
                const { error } = await supabase
                    .from('reservations')
                    .delete()
                    .eq('id', reservationId);
                
                if (error) throw error;
                
                alert('Réservation supprimée avec succès !');
                chargerReservations();
                chargerStatistiques();
                
            } catch (error) {
                console.error('Erreur lors de la suppression:', error);
                alert('Erreur lors de la suppression de la réservation');
            }
        }

        // Initialiser le dashboard
        document.addEventListener('DOMContentLoaded', () => {
            chargerStatistiques();
            chargerUtilisateurs();
            chargerReservations();
            
            // Recharger les données toutes les 30 secondes
            setInterval(() => {
                chargerStatistiques();
                chargerUtilisateurs();
                chargerReservations();
            }, 30000);
        });
    </script>
